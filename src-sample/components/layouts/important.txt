const handleFormSubmitPopUp = async () => {
  const isValid = await formMethods.trigger();
  
  if (!isValid) {
    const errors = formMethods.formState.errors;
    
    const pageFieldMap = {
      '/profile/information': ['firstName', 'lastName', 'profileUrl', 'bio'],
      '/profile/contact': ['whatsapp', 'email', 'text', 'isWhatsApp', 'isEmail', 'isText'],
    };
    
    const currentPageFields = pageFieldMap[pathname as keyof typeof pageFieldMap] || [];
    
    const hasCurrentPageErrors = currentPageFields.some(
      field => errors[field as keyof typeof errors]
    );
    
    if (hasCurrentPageErrors) {
      return;
    }
    
    const pageWithErrors = Object.entries(pageFieldMap).find(([path, fields]) => {
      if (path === pathname) return false;
      return fields.some(field => errors[field as keyof typeof errors]);
    });
    
    if (pageWithErrors) {
      router.push(pageWithErrors[0]);
    }
    
    return;
  }

  open("submit-confirmation", {
    title: "Are you sure you want to submit this form?",
    description: "",
    action: async () => {
      const formData = formMethods.getValues();
      await handleFormSubmit(formData);
      close();
    },
  });
};
